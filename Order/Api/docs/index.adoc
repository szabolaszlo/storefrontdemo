== UML ==

----
class Order {
    - id: string
    - customer: Customer
    - shippingAddress: ShippingAddress
    - billingAddress: BillingAddress
    - shippingMethod: ShippingMethod
    - billingMethod: BillingMethod
    - products: Product[]
    + getTotal(): number
}

class Product {
    - id: int
    - name: string
    - quantity: number
    - grossPrice: number
}

class ShippingAddress {
    - country: string
    - zipCode: number
    - city: string
    - address: string
}

class BillingAddress {
    - country: string
    - zipCode: number
    - city: string
    - address: string
}

class ShippingMethod {
    - code: string
    - grossPrice: number
}

class BillingMethod {
    - code: string
    - grossPrice: number
}

class Customer {
    - id: string
    - firstName: string
    - lastName: string
    - email: string
    - phoneNumber: string
}

== Rest API resources ==

=== POST /api/orders - creates a new order ===

Request:

[source]
----
{
    "checkoutId":"a198730c-e89e-4479-a8bd-587d4de6a68e"
}
----
Response:

[source]
----
{
    "orderId":"140f4f23-07d4-4feb-a9a7-9a22047b71bc"
}
----
